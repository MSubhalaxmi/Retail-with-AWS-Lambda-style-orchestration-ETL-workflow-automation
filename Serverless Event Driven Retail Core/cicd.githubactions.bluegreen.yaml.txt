name: Blue-Green Deployment

on:
  push:
    branches: [main]

jobs:
  deploy:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v2

      - name: Build Docker Image
        run: docker build -t retail-core .

      - name: Deploy Green
        run: echo "Deploying new version..."

      - name: Switch Traffic
        run: echo "Switching traffic to green environment"
